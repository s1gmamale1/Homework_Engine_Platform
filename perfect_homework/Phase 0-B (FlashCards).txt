  ┌─ KARTA 06 · Sodda Holat x² = d ──────────────┐
  │                                              │
  │  OLD TOMONI                                  │
  │       x² = d   →   x = ?                     │
  │                                              │
  │  ORQA TOMONI                                  │
  │       x₁,₂ = ±√d    (d ≥ 0 bo'lganda)        │
  │                                              │
  │  Uch holat:                                  │
  │                                              │
  │    d > 0      d = 0      d < 0               │
  │   ● ─── ●     ───●───    ─── ─── ───          │
  │  ikki ildiz   bitta ildiz   haqiqiy ildiz yo'q│
  │                                              │
  │  🧠 Yodlash usuli: parabola y = x² gorizontal  │
  │  d chizig'ini 2, 1 yoki 0 joyda kesadi.      │
  └──────────────────────────────────────────────┘

  ┌─ KARTA 07 · Kvadratlar Ayirmasi ─────────────┐
  │                                              │
  │  OLD TOMONI                                  │
  │       KVADRATLAR AYIRMASI                    │
  │       a² – b² = ?                            │
  │                                              │
  │  ORQA TOMONI                                  │
  │       a² – b² = (a – b)(a + b)               │
  │                                              │
  │  Quyidagilarni ko'paytuvchilarga ajratishda  │
  │  ishlatiladi:                                │
  │       x² – 64 = (x – 8)(x + 8)               │
  │       x² – d  = (x – √d)(x + √d)             │
  │                                              │
  │  🧠 Yodlash usuli: ikki kvadrat orasida bitta │
  │  minus belgisi — har doim yig'indi × ayirma   │
  │  ko'rinishida ajraladi.                      │
  └──────────────────────────────────────────────┘

  ┌─ KARTA 08 · Ko'paytuvchilarga Ajratish Andozasi ─┐
  │                                                  │
  │  OLD TOMONI                                      │
  │       x² + bx + c = ( ? )( ? )                   │
  │                                                  │
  │  ORQA TOMONI                                      │
  │  Quyidagi shartlarni qanoatlantiruvchi p, q     │
  │  sonlarini toping:                               │
  │                                                  │
  │       p + q  =  b   (o'rta koeffitsient)         │
  │       p · q  =  c   (ozod had)                   │
  │                                                  │
  │  Keyin:                                          │
  │       x² + bx + c = (x + p)(x + q)               │
  │                                                  │
  │  🧠 Yodlash usuli: "yig'indi = o'rta,            │
  │  ko'paytma = oxirgi." Ko'paytuvchi juftlarining  │
  │  qisqa jadvalini tuzing, to'g'ri juft yagona.    │
  └──────────────────────────────────────────────────┘
